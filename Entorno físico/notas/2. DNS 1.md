```bash
vim /etc/bind/named.conf.options
```
````shell
options {
	directory "/var/cache/bind";

	forwarders {
		8.8.8.8;
		8.8.4.4;
	};
	dnssec-validation auto;

	listen-on-v6 { any; };

	listen-on { 10.0.2.253; };

	querylog yes;

};
acl "trusted" {
	10.0.2.253;
	10.0.2.254;
};
````
